Using Fetch API:

async function getSummary(text) {
    try {
        const response = await fetch('https://aicrafters-summarize-api.vercel.app/api/summarize', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                text: text,
                engine: "azure"
            })
        });

        const data = await response.json();
        if (data.success) {
            return data.summary;
        } else {
            throw new Error(data.error);
        }
    } catch (error) {
        console.error('Error:', error);
        throw error;
    }
}

Using Requests:

import requests

def get_summary(text):
    url = "https://aicrafters-summarize-api.vercel.app/api/summarize"
    payload = {
        "text": text,
        "engine": "azure"
    }
    
    try:
        response = requests.post(url, json=payload)
        response.raise_for_status()
        data = response.json()
        
        if data["success"]:
            return data["summary"]
        else:
            raise Exception(data["error"])
    except requests.exceptions.RequestException as e:
        print(f"Error: {e}")
        raise

Using aiohttp (Async):

import aiohttp
import asyncio

async def get_summary_async(text):
    url = "https://aicrafters-summarize-api.vercel.app/api/summarize"
    payload = {
        "text": text,
        "engine": "azure"
    }
    
    async with aiohttp.ClientSession() as session:
        try:
            async with session.post(url, json=payload) as response:
                data = await response.json()
                if data["success"]:
                    return data["summary"]
                else:
                    raise Exception(data["error"])
        except Exception as e:
            print(f"Error: {e}")
            raise
